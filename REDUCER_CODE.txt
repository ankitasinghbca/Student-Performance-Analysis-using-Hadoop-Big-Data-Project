#!/usr/bin/env python
import sys
current_subject = ""
total = 0
count = 0
for line in sys.stdin:
      subject, marks = line.strip().split('\t')
      marks = float(marks)

      if subject == current_subject:
           total += marks
           count += 1
      else:
	       if current_subject != "":
                  print(current_subject + "\t" + str(total/count))
            current_subject = subject
            total = marks
            count = 1

if current_subject != "":
      print(current_subject + "\t" + str(total/count))
